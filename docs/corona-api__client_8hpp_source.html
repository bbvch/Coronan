125CountryData CoronaAPIClientType<ClientType>::request_country_data(std::string const& country_code,
126 std::optional<year_month_day> const& date) const
127{
128 auto const& date_query_string =
129 date.has_value() ? fmt::format("date={:%Y-%m-%d}&", sys_days(date.value())) : std::string{""};
130 auto const region_report_url = corona_api_url + std::string{"reports/total?"} + date_query_string +
131 std::string{"iso="} + std::string{country_code};
132
133 auto const http_response = ClientType::get(region_report_url);
134 if (http_response.status() == Poco::Net::HTTPResponse::HTTP_OK)
135 {
136 auto country_data = CountryData{};
137 country_data.info.iso_code = country_code;
138 if (auto const covid_data = coronan::api_parser::parse_region_total(http_response.response_body());
139 covid_data.has_value())
140 {
141 if (date.has_value())
142 {
143 country_data.timeline.emplace_back(covid_data.value());
144 }
145 else
146 {
147 country_data.latest = covid_data.value();
148 }
149 }
150 return country_data;
151 }
152 else
153 {
154 auto exception_msg = details::create_exception_msg(region_report_url, http_response);
155 throw HTTPClientException{std::move(exception_msg)};

159template <typename ClientType>
160CountryData CoronaAPIClientType<ClientType>::request_country_data(std::string const& country_code,
161 year_month_day const& start_date,
162 year_month_day const& end_date) const
163{
164 auto country_data = CountryData{};
165 country_data.info.iso_code = country_code;
166 auto const start = sys_days{start_date};
167 auto const end = sys_days{end_date};
168
169 std::vector<std::vector<sys_days>> days_to_parse;
170
171 std::vector<date::sys_days> days{};
172
173 for (auto day = start; day <= end; day += date::days{1})
174 {
175 days.emplace_back(day);
176 if (days.size() == details::number_of_parallel_get)
177 {
178 days_to_parse.emplace_back(days);
179 days = std::vector<date::sys_days>{};
180 }
181 }
182
183 if (not days.empty())
184 {
185 days_to_parse.emplace_back(days);
186 }
187
188 std::ranges::for_each(days_to_parse, [&](auto const& dates) {
189 using GetDataResult = std::variant<std::optional<CovidData>, std::string>;
190 std::vector<std::future<GetDataResult>> response_results;
191
192 for (auto const& day : dates)
193 {
194 auto const get_covid_data = [&]() -> GetDataResult {
195 auto const date = year_month_day{day};
196 auto const date_query_string = fmt::format("date={:%Y-%m-%d}&", date::sys_days(date));
197 auto const region_report_url = corona_api_url + std::string{"reports/total?"} + date_query_string +
198 std::string{"iso="} + std::string{country_code};
199
200 if (auto const http_response = ClientType::get(region_report_url);
201 http_response.status() == Poco::Net::HTTPResponse::HTTP_OK)
202 {
203 return coronan::api_parser::parse_region_total(http_response.response_body());
204 }
205 else
206 {
207 return details::create_exception_msg(region_report_url, http_response);
208 }
209 };
210
211 response_results.emplace_back(std::async(std::launch::async, get_covid_data));
212 };
213
214 for (auto& result : response_results)
215 {
216 auto const http_response = result.get();
217
218 std::visit(details::overloaded{[&](std::optional<CovidData> const& covid_data) {
219 if (covid_data.has_value())
220 {
221 country_data.timeline.emplace_back(covid_data.value());
222 }
223 },
224 [](std::string const& exception_msg) {
225 fmt::print(stderr, "HTTPClientException {}\n", exception_msg);
226 }},
227 http_response);
228 }
229 });
230 std::ranges::sort(country_data.timeline, [](auto const& lhs, auto const& rhs) { return lhs.date < rhs.date; });
231
232 return country_data;
233}
